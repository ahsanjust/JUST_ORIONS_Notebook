struct SuffixArray {
  vector<int> sa, rank, lcp;
  SuffixArray(const string &s, int lim = 256) {
    int n = s.size();
    sa.resize(n);
    rank.resize(n);
    vector<int> tmp(n), cnt(max(n, lim));
    for (int i = 0; i < n; i++)
      cnt[rank[i] = s[i]]++;
    for (int i = 1; i < lim; i++)
      cnt[i] += cnt[i - 1];
    for (int i = n - 1; i >= 0; i--)
      sa[--cnt[s[i]]] = i;
    for (int k = 1, r = 0; r < n - 1; k <<= 1,
    lim = r + 1) {
      int p = 0;
      for (int i = n - k; i < n; i++)
        tmp[p++] = i;
      for (int i = 0; i < n; i++)
        if (sa[i] >= k)
          tmp[p++] = sa[i] - k;
      fill(cnt.begin(), cnt.begin() + lim, 0);
      for (int i = 0; i < n; i++)
        cnt[rank[tmp[i]]]++;
      for (int i = 1; i < lim; i++)
        cnt[i] += cnt[i - 1];
      for (int i = n - 1; i >= 0; i--)
        sa[--cnt[rank[tmp[i]]]] = tmp[i];
      tmp[sa[0]] = r = 0;
      for (int i = 1; i < n; i++) {
        if (rank[sa[i]] != rank[sa[i - 1]] ||
        (sa[i] + k < n ? rank[sa[i] + k] : -1) !=
        (sa[i - 1] + k < n ? rank[sa[i - 1] + k] : -1))
          r++;
        tmp[sa[i]] = r;
      }
      rank = tmp;
    }
    lcp.assign(n, 0);
    vector<int> inv(n);
    for (int i = 0; i < n; i++) inv[sa[i]] = i;
    for (int i = 0, k = 0; i < n; i++) {
      if (inv[i] == 0) continue;
      int j = sa[inv[i] - 1];
      while (i + k < n && j + k < n && s[i + k] ==
      s[j + k]) k++;
      lcp[inv[i]] = k;
      if (k) k--;
    }
  }
};

int main() {
  string s = "banana";
  SuffixArray sa(s);
  for (int i : sa.sa) cout << i << ' ';
  cout << "\n";
  // LCP = Longest Common Prefix between sa[i]
  // and sa[i+1]
  for (int i = 1; i < s.size(); i++)
    cout << sa.lcp[i] << ' ';
}
