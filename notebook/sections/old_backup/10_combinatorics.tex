\subsection{Burnside's Lemma / Polya Enumeration}
\begin{minted}{c++}
using Permutation = vector<int>;
void operator*=(Permutation& p, Permutation
const& q) {
  Permutation copy = p;
  for (int i = 0; i < p.size(); i++)
    p[i] = copy[q[i]];
}
int count_cycles(Permutation p) {
  int cnt = 0;
  for (int i = 0; i < p.size(); i++) {
    if (p[i] != -1) {
      cnt++;
      for (int j = i; p[j] != -1;) {
        int next = p[j];
        p[j] = -1;
        j = next;
      }
    }
  }
  return cnt;
}
int solve(int n, int m) {
  Permutation p(n*m), p1(n*m), p2(n*m),
  p3(n*m);
  for (int i = 0; i < n*m; i++) {
    p[i] = i;
    p1[i] = (i % n + 1) % n + i / n * n;
    p2[i] = (i / n + 1) % m * n + i % n;
    p3[i] = (m - 1 - i / n) * n + (n - 1 - i % n);
  }
  set<Permutation> s;
  for (int i1 = 0; i1 < n; i1++) {
    for (int i2 = 0; i2 < m; i2++) {
      for (int i3 = 0; i3 < 2; i3++) {
        s.insert(p);
        p *= p3;
      }
      p *= p2;
    }
    p *= p1;
  }
  int sum = 0;
  for (Permutation const& p : s) {
    sum += 1 << count_cycles(p);
  }
  return sum / s.size();
}
\end{minted}
\subsection{Exclusion DP}
There is only difference to calculate $f[i]$ for different problems:

\textbf{For ordered pairs} $(a, b)$ where $(a, b) \neq (b, a)$: \\
$f[i] = d[i] \cdot d[i]$

\textbf{For unordered pairs} where $(a, b)$ and $(b, a)$ are the same: \\
\textbullet{} \textbf{With repetition}: $f[i] = d[i](d[i]+1)/2$ \\
\textbullet{} \textbf{No repetition}: $f[i] = d[i](d[i]-1)/2$

\textbf{For $k$-tuples}: \\
\textbullet{} \textbf{Ordered tuples}: $f[i] = (d[i])^k$ \\
\textbullet{} \textbf{Unordered (no rep)}: $f[i] = \binom{d[i]}{k}$ \\
\textbullet{} \textbf{Unordered (with rep)}: $f[i] = \binom{d[i]+k-1}{k}$ \\
Where $d[i]$ is the number of elements divisible by $i$.
\subsection{Bishop Placement}
\begin{minted}{c++}
int bishop_placements(int N, int K)
{
  if (K > 2 * N - 1)
    return 0;
  auto squares = [&](int i) {
    if (i > N) return 2 * N - i;
    return i;
  };
  vector<vector<int>> D(N * 2, vector<int>(K +
  1));
  for (int i = 0; i < N * 2; ++i)
    D[i][0] = 1;
  D[1][1] = 1;
  for (int i = 2; i < N * 2; ++i)
    for (int j = 1; j <= K; ++j)
      D[i][j] = D[i-2][j] + D[i-2][j-1] *
      (squares(i) - j + 1);
  int ans = 0;
  for (int i = 0; i <= K; ++i)
    ans += D[N*2-1][i] * D[N*2-2][K-i];
  return ans;
}
\end{minted}
\subsection{Combinatorics Formulas}

\textbf{Stirling Numbers of the 1st Kind $S_1(n,k)$}: \\
Permutations of $n$ elements with exactly $k$ cycles. \\
\textbullet{} \textbf{Recurrence}: $S_1(n,k) = S_1(n-1,k-1) + (n-1)S_1(n-1,k)$ \\
\textbullet{} \textbf{Total sum}: $\sum_{k=0}^n S_1(n,k) = n!$ \\
\textbullet{} \textbf{EGF}: $(-\ln(1-x))^k / k!$

\textbf{Stirling Numbers of the 2nd Kind $S_2(n,k)$}: \\
Ways to partition $n$ elements into $k$ non-empty sets. \\
\textbullet{} \textbf{Recurrence}: $S_2(n,k) = S_2(n-1,k-1) + k \cdot S_2(n-1,k)$ \\
\textbullet{} \textbf{Formula}: $S_2(n,k) = \frac{1}{k!} \sum_{i=0}^k (-1)^{k-i} \binom{k}{i} i^n$ \\
\textbullet{} \textbf{EGF}: $(\exp(x)-1)^k / k!$

\textbf{Bell Numbers $B(n)$}: \\
Total ways to partition a set of $n$ elements. \\
\textbullet{} \textbf{Relation}: $B(n) = \sum_{k=0}^n S_2(n,k)$ \\
\textbullet{} \textbf{Recurrence}: $B(n+1) = \sum_{k=0}^n \binom{n}{k} B(k)$ \\
\textbullet{} \textbf{EGF}: $\exp(\exp(x)-1)$

\textbf{Combinatorial Identities}: \\
\textbullet{} $\sum_{k=0}^n \binom{n-k}{k} = F_{n+1}$ (Also counts binary seqs of length $n$ without 00) \\
\textbullet{} Non-negative sols of $\sum x_i = n$ is $\binom{n+k-1}{n}$ \\
\textbullet{} \textbf{Pascal's}: $\binom{n}{k} + \binom{n}{k+1} = \binom{n+1}{k+1}$ \\
\textbullet{} \textbf{Absorption}: $k\binom{n}{k} = n\binom{n-1}{k-1}$

\textbf{Catalan Numbers $C_n$}: \\
Seq: 1, 1, 2, 5, 14, 42, 132, 429, 1430, 4862 \dots \\
\textbullet{} \textbf{Formula}: $C_n = \frac{1}{n+1} \binom{2n}{n} = \binom{2n}{n} - \binom{2n}{n+1}$ \\
\textbullet{} \textbf{Recurrence}: $C_{n+1} = \sum_{i=0}^n C_i C_{n-i}$ \\
\textbullet{} \textbf{OGF}: $\frac{1-\sqrt{1-4x}}{2x}$ \\
\textbullet{} \textbf{Applications}: Correct bracket seqs, polygon triangulations, non-crossing chords, paths $(0,0) \to (n,n)$ below diagonal, full binary trees, Dyck words.

\textbf{Eulerian Numbers $E(n,k)$}: \\
Permutations with exactly $k$ ascents (element $>$ previous). \\
\textbullet{} \textbf{Recurrence}: $E(n,k) = (n-k)E(n-1, k-1) + (k+1)E(n-1, k)$ \\
\textbullet{} \textbf{Formula}: $E(n,k) = \sum_{j=0}^k (-1)^j \binom{n+1}{j} (k+1-j)^n$ \\
\textbullet{} \textbf{Properties}: $E(n,k) = E(n, n-1-k)$

\textbf{Derangements $D(n)$}: \\
Permutations where $\pi(i) \neq i$. \\
Seq: 1, 0, 1, 2, 9, 44, 265, 1854 \dots \\
\textbullet{} \textbf{Recurrence}: $D(n) = (n-1)(D(n-1) + D(n-2))$ \\
\textbullet{} \textbf{Formula}: $D(n) = n! \sum_{k=0}^n \frac{(-1)^k}{k!} \approx \frac{n!}{e}$ \\
\textbullet{} \textbf{EGF}: $\frac{e^{-x}}{1-x}$

\textbf{Partition Numbers $P(n,k)$}: \\
Ways to write $n$ as a sum of exactly $k$ positive integers. \\
\textbullet{} \textbf{Recurrence}: $P(n,k) = P(n-1, k-1) + P(n-k, k)$ \\
\textbullet{} \textbf{Total partitions}: $P(n) = \sum_{k=0}^n P(n,k)$. (Seq: 1, 1, 2, 3, 5, 7, 11, 15, 22, 30 \dots)

\textbf{Generating Functions}: \\
\textbullet{} \textbf{OGF}: $A(x) = \sum_{i \ge 0} a_i x^i$ \\
\textbullet{} \textbf{Ops}: $xA'(x) \Rightarrow n a_n$, $A(x)B(x) \Rightarrow \sum_{i=0}^n a_i b_{n-i}$ \\
\textbullet{} \textbf{EGF}: $A(x) = \sum_{i \ge 0} a_i \frac{x^i}{i!}$ \\
\textbullet{} \textbf{Ops}: $A'(x) \Rightarrow a_{n+1}$, $xA(x) \Rightarrow n a_n$, $A(x)B(x) \Rightarrow \sum_{i=0}^n \binom{n}{i} a_i b_{n-i}$

\textbf{Bernoulli Numbers \& Faulhaber's}: \\
\textbullet{} \textbf{Recurrence}: $B_n = -\frac{1}{n+1} \sum_{k=0}^{n-1} \binom{n+1}{k} B_k$ \\
\textbullet{} \textbf{EGF}: $\frac{x}{e^x-1}$ \\
\textbullet{} \textbf{Sum of powers}: $\sum_{k=1}^n k^c = \frac{1}{c+1} \sum_{k=0}^c (-1)^k \binom{c+1}{k} B_k n^{c+1-k}$
